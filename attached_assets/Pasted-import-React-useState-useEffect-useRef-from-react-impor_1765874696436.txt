import React, { useState, useEffect, useRef } from 'react';
import { Copy, RotateCcw, Sparkles } from 'lucide-react';
import { Button } from '@/components/ui/button';

interface AnimatedMessageProps {
  content: string;
  role: 'user' | 'assistant';
  isStreaming?: boolean;
  onCopy?: (text: string) => void;
  onRegenerate?: () => void;
  thinkingTime?: number;
}

export function AnimatedMessage({
  content,
  role,
  isStreaming = false,
  onCopy,
  onRegenerate,
  thinkingTime,
}: AnimatedMessageProps) {
  const [displayedContent, setDisplayedContent] = useState('');
  const [isAnimating, setIsAnimating] = useState(isStreaming);
  const animationFrameRef = useRef<number>();
  const startTimeRef = useRef<number>();

  useEffect(() => {
    if (role === 'user') {
      // User messages appear instantly
      setDisplayedContent(content);
      return;
    }

    if (isStreaming) {
      // Streaming animation
      setIsAnimating(true);
      let index = 0;
      const intervalSpeed = 20; // milliseconds per character

      const animate = () => {
        if (index < content.length) {
          setDisplayedContent(content.slice(0, index + 1));
          index++;
          setTimeout(() => {
            animationFrameRef.current = requestAnimationFrame(animate);
          }, intervalSpeed);
        } else {
          setIsAnimating(false);
        }
      };

      animationFrameRef.current = requestAnimationFrame(animate);

      return () => {
        if (animationFrameRef.current) {
          cancelAnimationFrame(animationFrameRef.current);
        }
      };
    } else {
      setDisplayedContent(content);
    }
  }, [content, isStreaming, role]);

  const isUser = role === 'user';

  return (
    <div className={`flex gap-4 ${isUser ? 'justify-end' : 'justify-start'} group`}>
      {!isUser && (
        <div className="w-8 h-8 rounded-full bg-gradient-to-br from-green-400 to-blue-500 flex items-center justify-center flex-shrink-0">
          <Sparkles className="w-4 h-4 text-white" />
        </div>
      )}

      <div className={`max-w-[80%] ${isUser ? 'order-first' : ''}`}>
        <div
          className={`rounded-2xl px-4 py-3 ${
            isUser
              ? 'bg-gray-900 text-white ml-auto'
              : 'bg-gray-100 text-gray-900'
          }`}
        >
          <div className="whitespace-pre-wrap break-words text-[15px] leading-relaxed">
            {displayedContent}
            {isAnimating && (
              <span className="inline-block w-1 h-4 ml-1 bg-current animate-pulse" />
            )}
          </div>
        </div>

        {/* Action Buttons for Assistant Messages */}
        {!isUser && !isAnimating && (
          <div className="flex gap-2 mt-2 ml-1 opacity-0 group-hover:opacity-100 transition-opacity">
            {thinkingTime && (
              <span className="text-xs text-gray-400 self-center">
                Thought for {thinkingTime.toFixed(1)}s
              </span>
            )}
            {onCopy && (
              <Button
                variant="ghost"
                size="sm"
                className="h-7 px-2 text-xs hover:bg-gray-100"
                onClick={() => onCopy(content)}
              >
                <Copy className="w-3.5 h-3.5 mr-1" />
                Copy
              </Button>
            )}
            {onRegenerate && (
              <Button
                variant="ghost"
                size="sm"
                className="h-7 px-2 text-xs hover:bg-gray-100"
                onClick={onRegenerate}
              >
                <RotateCcw className="w-3.5 h-3.5 mr-1" />
                Regenerate
              </Button>
            )}
          </div>
        )}
      </div>

      {isUser && (
        <div className="w-8 h-8 rounded-full bg-gray-900 flex items-center justify-center flex-shrink-0">
          <span className="text-white text-sm font-medium">U</span>
        </div>
      )}
    </div>
  );
}

// Real-time streaming component
export function StreamingMessage({ onChunk }: { onChunk?: (text: string) => void }) {
  const [streamedContent, setStreamedContent] = useState('');
  const [isComplete, setIsComplete] = useState(false);

  // Simulate streaming (replace with actual SSE implementation)
  useEffect(() => {
    const words = [
      'Hello!',
      'I',
      'can',
      'help',
      'you',
      'with',
      'various',
      'tasks',
      'today.',
      'What',
      'would',
      'you',
      'like',
      'to',
      'know?',
    ];

    let index = 0;
    let accumulated = '';

    const interval = setInterval(() => {
      if (index < words.length) {
        accumulated += (index > 0 ? ' ' : '') + words[index];
        setStreamedContent(accumulated);
        onChunk?.(accumulated);
        index++;
      } else {
        setIsComplete(true);
        clearInterval(interval);
      }
    }, 150);

    return () => clearInterval(interval);
  }, []);

  return (
    <AnimatedMessage
      content={streamedContent}
      role="assistant"
      isStreaming={!isComplete}
    />
  );
}
