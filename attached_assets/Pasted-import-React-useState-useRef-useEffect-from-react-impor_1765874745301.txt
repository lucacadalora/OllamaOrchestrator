import React, { useState, useRef, useEffect } from 'react';
import { Send, Sparkles, Layers, ChevronDown } from 'lucide-react';
import { Button } from '@/components/ui/button';
import { Textarea } from '@/components/ui/textarea';
import { Badge } from '@/components/ui/badge';
import { AnimatedMessage, ThinkingIndicator, ProcessingOverlay, InferenceGlow } from './AnimatedMessage';
import { WorldMap } from './WorldMap';

interface Message {
  id: string;
  role: 'user' | 'assistant';
  content: string;
  timestamp: Date;
  thinkingTime?: number;
  processingNodeId?: string;
}

interface NodeLocation {
  id: string;
  name: string;
  lat: number;
  lng: number;
  status: 'active' | 'idle' | 'offline';
  model?: string;
}

interface CompleteChatInterfaceProps {
  modelName: string;
  nodes: NodeLocation[];
  onSendMessage: (message: string) => Promise<{ response: string; nodeId?: string }>;
  onShowBackend?: () => void;
}

export function CompleteChatInterface({
  modelName,
  nodes,
  onSendMessage,
  onShowBackend,
}: CompleteChatInterfaceProps) {
  const [messages, setMessages] = useState<Message[]>([]);
  const [inputValue, setInputValue] = useState('');
  const [isStreaming, setIsStreaming] = useState(false);
  const [currentProcessingNode, setCurrentProcessingNode] = useState<string | undefined>();
  const [showInferenceJobs, setShowInferenceJobs] = useState(false);
  const messagesEndRef = useRef<HTMLDivElement>(null);
  const textareaRef = useRef<HTMLTextAreaElement>(null);

  const scrollToBottom = () => {
    messagesEndRef.current?.scrollIntoView({ behavior: 'smooth' });
  };

  useEffect(() => {
    scrollToBottom();
  }, [messages]);

  const handleSend = async () => {
    if (!inputValue.trim() || isStreaming) return;

    const userMessage: Message = {
      id: Date.now().toString(),
      role: 'user',
      content: inputValue,
      timestamp: new Date(),
    };

    setMessages(prev => [...prev, userMessage]);
    setInputValue('');
    setIsStreaming(true);

    const startTime = Date.now();

    try {
      const { response, nodeId } = await onSendMessage(inputValue);
      const thinkingTime = (Date.now() - startTime) / 1000;

      setCurrentProcessingNode(nodeId);

      // Simulate brief delay to show processing animation
      await new Promise(resolve => setTimeout(resolve, 500));

      const assistantMessage: Message = {
        id: (Date.now() + 1).toString(),
        role: 'assistant',
        content: response,
        timestamp: new Date(),
        thinkingTime,
        processingNodeId: nodeId,
      };

      setMessages(prev => [...prev, assistantMessage]);
      setCurrentProcessingNode(undefined);
    } catch (error) {
      console.error('Error sending message:', error);
    } finally {
      setIsStreaming(false);
    }
  };

  const handleKeyDown = (e: React.KeyboardEvent) => {
    if (e.key === 'Enter' && !e.shiftKey) {
      e.preventDefault();
      handleSend();
    }
  };

  const copyToClipboard = (text: string) => {
    navigator.clipboard.writeText(text);
  };

  const activeNodes = nodes.filter(n => n.status === 'active');

  return (
    <div className="flex flex-col h-screen bg-gradient-to-b from-gray-50 to-white relative">
      {/* Processing animations */}
      {isStreaming && (
        <>
          <ProcessingOverlay nodeId={currentProcessingNode} />
          <InferenceGlow />
        </>
      )}

      {/* Header */}
      <div className="border-b bg-white/80 backdrop-blur-sm sticky top-0 z-10">
        <div className="max-w-4xl mx-auto px-6 py-4 flex items-center justify-between">
          <div className="flex items-center gap-3">
            <div className="w-8 h-8 bg-gradient-to-br from-green-400 to-blue-500 rounded-lg flex items-center justify-center">
              <Sparkles className="w-5 h-5 text-white" />
            </div>
            <div>
              <h1 className="text-xl font-semibold">Welcome to DGON.</h1>
              <p className="text-sm text-gray-500">Powered by {modelName}</p>
            </div>
          </div>
          <Button
            variant="outline"
            size="sm"
            className="gap-2"
            onClick={() => setShowInferenceJobs(!showInferenceJobs)}
          >
            <Layers className="w-4 h-4" />
            Explore Network
          </Button>
        </div>
      </div>

      {/* Messages Container */}
      <div className="flex-1 overflow-y-auto">
        <div className="max-w-4xl mx-auto px-6 py-8">
          {messages.length === 0 ? (
            <EmptyState modelName={modelName} nodes={nodes} />
          ) : (
            <div className="space-y-6">
              {messages.map((message, index) => (
                <AnimatedMessage
                  key={message.id}
                  content={message.content}
                  role={message.role}
                  isStreaming={
                    index === messages.length - 1 &&
                    message.role === 'assistant' &&
                    isStreaming
                  }
                  onCopy={message.role === 'assistant' ? copyToClipboard : undefined}
                  thinkingTime={message.thinkingTime}
                />
              ))}
              {isStreaming && messages[messages.length - 1]?.role === 'user' && (
                <ThinkingIndicator />
              )}
            </div>
          )}
          <div ref={messagesEndRef} />
        </div>
      </div>

      {/* Collapsible Inference Jobs Section */}
      {showInferenceJobs && (
        <div className="border-t bg-white">
          <button
            onClick={() => setShowInferenceJobs(false)}
            className="w-full flex items-center justify-center gap-2 py-3 text-sm font-medium text-gray-600 hover:text-gray-900 transition-colors"
          >
            <span>Inference Jobs</span>
            <ChevronDown className="w-4 h-4" />
          </button>
          <div className="max-w-4xl mx-auto px-6 pb-4">
            <WorldMapSection nodes={nodes} activeNodeId={currentProcessingNode} />
          </div>
        </div>
      )}

      {/* Input Area */}
      <div className="border-t bg-white sticky bottom-0">
        <div className="max-w-4xl mx-auto px-6 py-4">
          <div className="relative">
            <div className="flex gap-3 items-end bg-white border rounded-2xl shadow-lg p-3 focus-within:ring-2 focus-within:ring-green-500 focus-within:border-transparent transition-all">
              {/* Action Buttons */}
              <div className="flex gap-2 pb-2">
                <Button
                  variant="ghost"
                  size="icon"
                  className="rounded-xl hover:bg-gray-100"
                  title="Upload image"
                >
                  <svg className="w-5 h-5 text-gray-600" viewBox="0 0 24 24" fill="currentColor">
                    <path d="M21 19V5c0-1.1-.9-2-2-2H5c-1.1 0-2 .9-2 2v14c0 1.1.9 2 2 2h14c1.1 0 2-.9 2-2zM8.5 13.5l2.5 3.01L14.5 12l4.5 6H5l3.5-4.5z"/>
                  </svg>
                </Button>
                <Button
                  variant="ghost"
                  size="icon"
                  className="rounded-xl hover:bg-gray-100"
                  title="Attach file"
                >
                  <svg className="w-5 h-5 text-gray-600" viewBox="0 0 24 24" fill="none" stroke="currentColor" strokeWidth="2">
                    <path d="M21.44 11.05l-9.19 9.19a6 6 0 01-8.49-8.49l9.19-9.19a4 4 0 015.66 5.66l-9.2 9.19a2 2 0 01-2.83-2.83l8.49-8.48"/>
                  </svg>
                </Button>
              </div>

              {/* Textarea */}
              <Textarea
                ref={textareaRef}
                value={inputValue}
                onChange={(e) => setInputValue(e.target.value)}
                onKeyDown={handleKeyDown}
                placeholder="How can I help you today?"
                className="flex-1 border-0 resize-none focus-visible:ring-0 focus-visible:ring-offset-0 min-h-[44px] max-h-[200px] text-base"
                rows={1}
                disabled={isStreaming}
              />

              {/* Thinking Badge */}
              {isStreaming && (
                <Badge variant="secondary" className="gap-1.5 self-center">
                  <Sparkles className="w-3 h-3 animate-pulse" />
                  Thinking
                </Badge>
              )}

              {/* Model Selector */}
              <select className="self-center px-3 py-2 text-sm border-0 bg-transparent text-gray-700 font-medium cursor-pointer hover:bg-gray-50 rounded-lg transition-colors">
                <option>{modelName}</option>
              </select>

              {/* Send Button */}
              <Button
                onClick={handleSend}
                disabled={!inputValue.trim() || isStreaming}
                size="icon"
                className="rounded-xl bg-green-500 hover:bg-green-600 disabled:bg-gray-200 disabled:text-gray-400 transition-all self-end"
              >
                <Send className="w-5 h-5" />
              </Button>
            </div>
          </div>

          {/* Helper Text */}
          <p className="text-xs text-gray-400 text-center mt-3">
            DGON can make mistakes. Consider checking important information.
          </p>
        </div>
      </div>
    </div>
  );
}

function WorldMapSection({ nodes, activeNodeId }: { nodes: NodeLocation[]; activeNodeId?: string }) {
  return (
    <div className="bg-gray-50/50 rounded-xl p-6 border border-gray-200">
      <div className="flex items-center justify-between mb-4">
        <div>
          <h3 className="font-semibold text-lg">Inference Jobs</h3>
          <p className="text-sm text-gray-500">
            {nodes.filter(n => n.status === 'active').length} active nodes worldwide
          </p>
        </div>
        {activeNodeId && (
          <Badge variant="secondary" className="gap-1.5">
            <div className="w-2 h-2 rounded-full bg-green-500 animate-pulse" />
            Processing
          </Badge>
        )}
      </div>
      <div className="h-48">
        <WorldMap nodes={nodes} activeNodeId={activeNodeId} />
      </div>
    </div>
  );
}
